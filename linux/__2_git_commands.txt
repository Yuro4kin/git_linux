# директории и файлы

		l - long
		a - all
ls -la          показать еще скрытые директории     
drwxr-xr-x 1 User 197121  0 Feb 10 16:43 ./      - одна точка, где сейчас нахожусь - текущая директория
drwxr-xr-x 1 User 197121  0 Feb 10 11:50 ../     - две точки - это директория выше

LINUX: 
      ~ cd home 
      ~ cd Documents/    какие у нас есть документы

pwd - print working directory - команда где я нахожусь сейчас - /c/Users/User

cd .. выйти на одну директорию вверх

cd ../.. выйти на две дирректории вверх, вначало

cd /     выйти в home самое начало

cd c:/Users/User                      вышел в home директорию
User@DESKTOP-LSQSTCM MINGW64 ~

cd /etc               я иду в etc в самом начале системы etc

cd                    я попадаю в home - домашнюю директорию
cd ~                  я попадаю в home - аналог команда

 cd ~/Documents/      сразу попадаю в папку документы

ls -a                 покажет скрытые файлы

ls -la -R /           покажет все файлы из всех директорий

start "gmail.py"      открытие файла из командной строки
start cd "gmail.py"
